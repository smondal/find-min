## Install

```
$ npm install @sandipmo/find-min
```


## Usage

```js
const findMin = require('find-min');

var max = findMin(1, 2, 3, 4)  // 4

var max = findMin([1, 2, 3, 4]) //4

var max = findMin({1: 2, 2: 4, 4: 5})  //'4'
```
